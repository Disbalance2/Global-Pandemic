		available = {
			has_war = no

				type = Inf_equipment

			}

				random_list = {

					10 = { add_to_variable = { THIS.kasmir_readiness = 20 } }
				}

				clamp_variable = {
					var = THIS.kasmir_readiness
					min = 0
					max = 100
				}
			}
		}

		ai_will_do = {
			factor = 5
			modifier = {
				factor = 0
				threat < 0.5
			}
			modifier = {
				factor = 0
				check_variable = { THIS.kasmir_readiness > 80 }
			}
		}
	}
