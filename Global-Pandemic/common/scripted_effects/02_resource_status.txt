status_energy_resource = {
	clear_variable = energy_resource_status
		every_controlled_state = {
		add_to_temp_variable = { energy_resource_temp_status = building_level@thermal_power_plant }
		add_to_temp_variable = { energy_resource_temp_status = building_level@arms_factory }
		add_to_temp_variable = { energy_resource_temp1_status = building_level@industrial_complex }
		add_to_temp_variable = { energy_resource_temp_status = building_level@steel_factory }
		add_to_temp_variable = { energy_resource_temp_status = building_level@aluminium_factory }
		add_to_temp_variable = { energy_resource_temp_status = building_level@alloy_steel_factory }
		add_to_temp_variable = { energy_resource_temp_status = building_level@rare_earth_factory }
		add_to_temp_variable = { energy_resource_temp_status = building_level@uran_factory }
	}
	multiply_temp_variable = { energy_resource_temp1_status = 2 }
	add_to_variable = { energy_resource_status = resource_consumed@energy_resource }
	add_to_variable = { energy_resource_status = energy_resource_temp_status }
}
