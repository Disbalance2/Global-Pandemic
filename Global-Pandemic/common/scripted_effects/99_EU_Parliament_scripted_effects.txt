election_EU_Parliament = {
	### calculate MEP per member state
	set_temp_variable = { temp1 = 0 }
	#for_each_loop = { ### seems not to work in this case
	#	array = global.EU_member
	every_country = {
		limit = {
			has_idea = EU_member
		}
		add_to_temp_variable = { temp1 = 1 }
	}
	multiply_temp_variable = { temp1 = 6 }
	set_temp_variable = { temp2 = 750 }
	subtract_from_temp_variable = { temp2 = temp1 }
	set_variable = { global.EU_Parliament_per_EUpopRatio = temp2 }
	### Election in member_state
	#for_each_loop = { ### seems not to work in this case
	#	array = global.EU_member
	every_country = {
		limit = {
			has_idea = EU_member
		}
		election_EU_Parliament_in_member_state = yes
	}
	set_variable = { global.MEP_Majority = global.MEP_Total }
	divide_variable = { global.MEP_Majority = 2 }
	add_to_variable = { global.MEP_Majority = 1 }
	set_variable = { global.MEP_Supermajority = global.MEP_Total }
	divide_variable = { global.MEP_Supermajority = 3 }
	multiply_variable = { global.MEP_Supermajority = 2 }
	add_to_variable = { global.MEP_Supermajority = 1 }

	### check political groups
	#Western_Autocracy
	if = {
		limit = {
			check_variable = { global.MEP_PG_Western_Autocracy > 24 }
		}
		set_global_flag = EU_parliament_PG_Western_Autocracy
		set_variable = { global.MEP_Ratio_PG_Western_Autocracy = global.MEP_PG_Western_Autocracy }
		divide_variable = { global.MEP_Ratio_PG_Western_Autocracy = global.MEP_Total }
		add_to_array = { global.MEP_PG_array^0 = global.MEP_PG_Western_Autocracy }
		set_variable = { global.PI_PG_Western_Autocracy = global.MEP_PG_Western_Autocracy }
	}
	else = {
		every_country = {
			limit = {
				has_idea = EU_member
				check_variable = { THIS.MEP_Western_Autocracy > 0 }
			}
			set_country_flag = EU_parliament_IND_Western_Autocracy
		}
		set_variable = { global.MEP_IND_Western_Autocracy = global.MEP_PG_Western_Autocracy }
		set_variable = { global.MEP_Ratio_IND_Western_Autocracy = global.MEP_IND_Western_Autocracy }
		divide_variable = { global.MEP_Ratio_IND_Western_Autocracy = global.MEP_Total }
		clear_variable = global.MEP_PG_Western_Autocracy
		add_to_array = { global.MEP_IND_array^0 = global.MEP_IND_Western_Autocracy }
	}
	#conservatism

		limit = {
			check_variable = { global.MEP_PG_conservatism > 24 }
