				}

					PREV = { AC_building_antiair_available = yes }
				}
				else_if = { limit = { check_variable = { project_building_type^-1 = 7 } }

					PREV = { AC_building_radar_available = yes }
				}
				else_if = { limit = { check_variable = { project_building_type^-1 = 8 } }

					PREV = { AC_building_airbase_available = yes }
				}
				else_if = { limit = { check_variable = { project_building_type^-1 = 9 } }
					PREV = { building_fuel_silos_available = yes }
				}
				else_if = { limit = { check_variable = { project_building_type^-1 = 10 } }
					PREV = { building_internet_station_available = yes }
				}
				else_if = { limit = { check_variable = { project_building_type^-1 = 11 } }
					PREV = { AC_building_ref_available = yes }
				}
				else_if = { limit = { check_variable = { project_building_type^-1 = 12 } }
					PREV = { AC_building_foss_fuel_powerplant_available = yes }
				}
				else_if = { limit = { check_variable = { project_building_type^-1 = 13 } }
					PREV = { AC_building_reactor_available = yes }
				}
			}
		}
		localization_key = "AC_build_amount_green"
	}
	text = {
		localization_key = "AC_build_amount_red"
	}
}
