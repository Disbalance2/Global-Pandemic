		AND = {

			GER = {
				has_navy_size = {
					type = submarine
					size > 30
				}
			}
		}
		AND = {
			tag = USA

			GER = {
				has_navy_size = {
					type = submarine
					size > 30
				}

			not = { is_in_faction_with = USA }
