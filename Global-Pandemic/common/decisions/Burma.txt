			else_if = { limit = { hostile_the_priesthood = yes }
				random_list = {
					10 = { change_the_priesthood_opinion = yes }
					90 = { add_political_power = -100 }
				}
			}
			else_if = { limit = { positive_the_priesthood = yes }
				random_list = {
					30 = { change_the_priesthood_opinion = yes }
					70 = { add_political_power = -100 }
				}
			}
			else_if = { limit = { enthusiastic_the_priesthood = yes }
				add_political_power = 50
			}
		}

		ai_will_do = {
			factor = 1
		}
	}
}
