		fire_only_once = no
		days_remove = 90
		days_re_enable = 150 # Let's have some balancing checks, should re-enable 5 months after complete

		visible = {
			check_variable = { next_slot_from_gdp < 10 }
		}

		available = {
			custom_trigger_tooltip = {
