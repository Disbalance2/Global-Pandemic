#Скрипт что заставляет ИИ строить энергостанции при нехватке электроэнергии и доступности ресурсов

AI_build_max_priority_thermal_power_plant = {
	enable = {
		check_variable = { electricity_ratio > 25 }
		check_variable = { resource@energy_resource > 4 }
	}
	
	abort = {
		check_variable = { electricity_ratio < 30 }
		check_variable = { resource@energy_resource < 4 }
	}
	
	ai_strategy = {
		type = building_target
		id = thermal_power_plant		
		value = 5
	}
}
