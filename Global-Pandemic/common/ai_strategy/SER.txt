SER_hate_KOS = {
	allowed = { original_tag = SER }
	enable = {
		country_exists = KOS
	}
	abort = {
		KOS = { is_subject_of = SER }
	}

	ai_strategy = { type = antagonize id = "KOS" value = 50 }
	ai_strategy = { type = conquer id = "KOS" value = 50 }
}

SER_befriend_SOV = {
	allowed = { original_tag = SER }
	enable = {
		SOV = { NOT = { has_government = democratic } }
		NOT = { has_government = democratic }
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = befriend id = "SOV" value = 50 }
}
